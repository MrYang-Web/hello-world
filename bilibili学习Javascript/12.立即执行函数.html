12.立即执行函数

        //立即执行函数:
        //当一个函数并不需要声明,只需要简单被调用一次的时候.
        //可以使用立即执行函数,
        //函数被创建的时候,就会马上执行.
        //语法:   (function(){
                      代码语句....
                  });
        (function () {
          alert("hello world!");
        });
        (function (a,b) {
          console.log("a =" + a);
          console.log("b = "+ b);
        })(1111,2222);
