jQuery的map方法

      var arr = [1, 3, 5, 7, 9,];
      var obj = {0:1, 1:3, 2:5, 3:7, 4:9, length:5};
      // 1.利用原生JS进行遍历
      // 第一个value 是当前遍历到的元素
      // 第二个index 是当前遍历到的索引
      // 第三个array 是当前遍历到的数组
      //注意点:和原生forEach一样,不能遍历伪数组
      // arr.map(function (value,index,array) {
      //   console.log(index,value,array);
      // })
      //----------------------------------------
      // 第一个值arr 是当前要遍历到的的数组
      // 第二个function 是每遍历一个元素所执行的回调函数
      //     回调函数的值:  第一个value 是遍历到的元素
      //                   第二个index 是遍历到的索引
      //注意点:和jQuery的Each方法一样,map方法也能遍历伪数组
      // $.map(arr,function (value,index) {
      //   console.log(index,value);
      // });
      // $.map(obj,function (value,index) {
      //   console.log(index,value);
      // });


      var res = $.map(arr,function (value,index) {
        console.log(index,value);
        return value + index;  //支持
      });
			
      var res2 = $.each(obj,function (value,index) {
        console.log(index,value);
        return value + index;  //不支持
      });

      console.log(res);
      console.log(res2);
      // jQuery当中map和each静态方法的区别:
      //  map方法默认返回的是一个空数组.
      //  each方法默认返回的是遍历谁就返回谁.
      //
      //  map方法支持在回调函数中使用retur对遍历的数组处理生成一个新数组返回.
      //  each方法不支持在回调函数用return对遍历的数组进行处理.
